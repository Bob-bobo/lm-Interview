## 🧹 三、数据清洗与质量控制

### 1️⃣ 清洗脚本示例

```python
# data_cleaning.py
import json
import re
from typing import List, Dict

class DataCleaner:
    def __init__(self):
        self.min_length = 10
        self.max_length = 2048
        
    def remove_special_chars(self, text: str) -> str:
        """移除特殊字符和乱码"""
        text = re.sub(r'[^\w\s\u4e00-\u9fff,.!?]', '', text)
        return text.strip()
    
    def filter_short_samples(self, data: List[Dict]) -> List[Dict]:
        """过滤过短样本"""
        return [d for d in data if len(d.get('output', '')) > self.min_length]
    
    def remove_duplicates(self, data: List[Dict]) -> List[Dict]:
        """去重（基于 instruction+output）"""
        seen = set()
        unique_data = []
        for d in 
            key = f"{d.get('instruction', '')}{d.get('output', '')}"
            if key not in seen:
                seen.add(key)
                unique_data.append(d)
        return unique_data
    
    def detect_noise(self, text: str) -> bool:
        """检测噪声（乱码、重复字符等）"""
        if len(text) == 0:
            return True
        if text.count(text[0]) / len(text) > 0.5:  # 重复字符过多
            return True
        if '\x00' in text or '\ufffd' in text:  # 乱码字符
            return True
        return False
    
    def clean_dataset(self, input_file: str, output_file: str):
        """完整清洗流程"""
        with open(input_file, 'r', encoding='utf-8') as f:
            data = [json.loads(line) for line in f]
        
        print(f"原始数据量：{len(data)}")
        
        # 清洗步骤
        for d in 
            d['instruction'] = self.remove_special_chars(d.get('instruction', ''))
            d['output'] = self.remove_special_chars(d.get('output', ''))
        
        data = self.filter_short_samples(data)
        print(f"过滤短样本后：{len(data)}")
        
        data = self.remove_duplicates(data)
        print(f"去重后：{len(data)}")
        
        data = [d for d in data if not self.detect_noise(d.get('output', ''))]
        print(f"去噪声后：{len(data)}")
        
        # 保存
        with open(output_file, 'w', encoding='utf-8') as f:
            for d in data:
                f.write(json.dumps(d, ensure_ascii=False) + '\n')
        
        print(f"✅ 清洗完成，保存至 {output_file}")

# 使用
cleaner = DataCleaner()
cleaner.clean_dataset('raw_data.jsonl', 'cleaned_data.jsonl')
```

### 2️⃣ 质量评估指标

| 指标 | 计算方法 | 合格标准 |
|------|---------|---------|
| **重复率** | 重复样本数/总样本数 | <5% |
| **噪声率** | 含乱码/过短样本数/总样本数 | <3% |
| **长度分布** | instruction/output 长度统计 | 符合模型上下文窗口 |
| **多样性** | 词汇丰富度、主题覆盖度 | 覆盖目标场景 90%+ |
| **一致性** | 多标注者 Kappa 系数 | >0.8 |

```python
# convert_to_hf_dataset.py
from datasets import load_dataset, Dataset
import json

def convert_jsonl_to_hf(input_file: str, output_dir: str):
    """JSONL → HuggingFace Dataset"""
    
    # 加载数据
    data = []
    with open(input_file, 'r', encoding='utf-8') as f:
        for line in f:
            data.append(json.loads(line))
    
    # 转换为 Dataset
    dataset = Dataset.from_list(data)
    
    # 划分训练/验证集
    dataset_dict = dataset.train_test_split(test_size=0.1)
    
    # 保存
    dataset_dict.save_to_disk(output_dir)
    print(f"✅ 数据集保存至 {output_dir}")
    
    # 统计信息
    print(f"训练集：{len(dataset_dict['train'])} 样本")
    print(f"验证集：{len(dataset_dict['test'])} 样本")

# 使用
convert_jsonl_to_hf('cleaned_data.jsonl', './hf_dataset')
```

### 2️⃣ 昇腾 NPU 适配数据加载（参考项目一）

```python
# npu_dataloader.py
import torch
from torch.utils.data import DataLoader, Dataset
import torch_npu
from torch_npu.contrib import transfer_to_npu

class SFTDataset(Dataset):
    def __init__(self, data_path, tokenizer, max_length=512):
        self.data = []
        with open(data_path, 'r', encoding='utf-8') as f:
            for line in f:
                self.data.append(json.loads(line))
        self.tokenizer = tokenizer
        self.max_length = max_length
    
    def __len__(self):
        return len(self.data)
    
    def __getitem__(self, idx):
        item = self.data[idx]
        text = f"### Instruction:\n{item['instruction']}\n\n### Response:\n{item['output']}"
        
        encoding = self.tokenizer(
            text,
            truncation=True,
            max_length=self.max_length,
            padding='max_length',
            return_tensors='pt'
        )
        
        return {
            'input_ids': encoding['input_ids'].squeeze(0),
            'attention_mask': encoding['attention_mask'].squeeze(0),
            'labels': encoding['input_ids'].squeeze(0).clone()
        }

# NPU 优化 DataLoader
def get_npu_dataloader(dataset, batch_size=4):
    return DataLoader(
        dataset,
        batch_size=batch_size,
        shuffle=True,
        num_workers=4,  # 昇腾推荐值
        pin_memory=True,  # 加速 CPU→NPU 传输
        persistent_workers=True
    )
```

### 3️⃣ 多模态数据加载（参考项目二）

```python
# multimodal_dataloader.py
import torch
from torch.utils.data import Dataset
from PIL import Image
import torchaudio

class MultimodalDataset(Dataset):
    def __init__(self, data_path, image_processor, audio_processor, tokenizer):
        with open(data_path, 'r') as f:
            self.data = [json.loads(line) for line in f]
        self.image_processor = image_processor
        self.audio_processor = audio_processor
        self.tokenizer = tokenizer
    
    def __len__(self):
        return len(self.data)
    
    def __getitem__(self, idx):
        item = self.data[idx]
        
        # 图像处理
        image = Image.open(item['image_path']).convert('RGB')
        image_tensor = self.image_processor(image, return_tensors='pt')['pixel_values'].squeeze(0)
        
        # 音频处理（项目二音乐生成场景）
        audio, sr = torchaudio.load(item['audio_path'])
        audio_tensor = self.audio_processor(audio, sr, return_tensors='pt')['input_features'].squeeze(0)
        
        # 文本处理
        text_encoding = self.tokenizer(
            item['text'],
            truncation=True,
            max_length=256,
            return_tensors='pt'
        )
        
        return {
            'image': image_tensor,
            'audio': audio_tensor,
            'input_ids': text_encoding['input_ids'].squeeze(0),
            'attention_mask': text_encoding['attention_mask'].squeeze(0),
            'label': item.get('label', 0)
        }
```

## ⚠️ 六、常见问题与解决方案

| 问题 | 原因 | 解决方案 |
|------|------|----------|
| **数据格式错误** | JSON 格式不规范、编码问题 | 统一使用 UTF-8，用 `json.loads()` 验证 |
| **显存溢出** | 序列过长、batch_size 过大 | 限制 max_length，启用 gradient_accumulation |
| **训练不收敛** | 数据质量差、标注不一致 | 加强清洗，人工抽样检查 100+ 样本 |
| **过拟合** | 数据量太少、重复率高 | 增加数据增强，降低学习率，早停 |
| **NPU 加载慢** | 数据预处理在 CPU、IO 瓶颈 | 使用 `num_workers>0`，`pin_memory=True` |
| **多模态对齐失败** | 图文/文音不匹配 | 加强数据标注审核，使用 CLIP Score 自动过滤 |
